\(\| A + B \|_{\mu,\nu} \leq \| A \|_{\mu,\nu} + \| B \|_{\mu,\nu} \) (\(\| \cdot \|_{\mu,\nu} \) obeys the triangle inequality).

\begin{equation*} \begin{array}{l} \| A + B \|_{\mu,\nu} \\ ~~~=~~~~ \lt {\rm definition} \gt \\ \max_{x \neq 0} } \frac{\| ( A + B )x \|_\mu}{\| x \|_{\nu}} \\ ~~~=~~~~ \lt {\rm distributivity} \gt \\ \max_{x \neq 0} \frac{\| Ax + B x \|_\mu}{\| x \|_{\nu}} \\ ~~~\le~~~~ \lt {\rm triangle~inequality} \gt \\ \max_{x \neq 0} \frac{\| Ax \|_\mu + \| B x \|_\mu}{\| x \|_{\nu}} \\ ~~~\le~~~~ \lt {\rm algebra} \gt \\ \max_{x \neq 0} \left( \frac{\| Ax \|_\mu}{\| x \|_{\nu}} + \frac{\| B x \|_\mu}{\| x \|_{\nu}} \right) \\ ~~~\leq~~~~ \lt {\rm algebra} \gt \\ \max_{x \neq 0} \frac{\| Ax \|_\mu}{\| x \|_{\nu}} + \max_{x \neq 0} \frac{\| B x \|_\mu}{\| x \|_{\nu}} \\ ~~~=~~~~ \lt {\rm definition} \gt \\ \| A \|_{\mu,\nu} + \| B \|_{\mu,\nu} . \end{array} \end{equation*}
